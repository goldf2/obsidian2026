## 1. 需求背景

### 1.1 现状问题

当前系统在权限分配时存在以下问题：

1. **用户列表过长**：随着用户增加，在权限分配界面查找用户困难
2. **无部门筛选**：无法按部门快速定位用户
3. **部门配置分散**：IATF 使用硬编码的 `DEPT_CONFIG`，不便维护
4. **缺乏部门归属**：用户表 `auth_user` 无部门字段

### 1.2 目标

1. 建立统一的部门管理机制
2. 用户关联所属部门
3. 权限分配界面支持按部门筛选
4. 为后续功能（部门统计、组织架构图等）奠定基础

### 1.3 影响范围

| 应用 | 影响模块 | 说明 |
|------|----------|------|
| service_center | 用户管理 | 添加部门字段、部门筛选 |
| service_center | 部门管理 | 新增功能模块 |
| IATF | 权限管理 | 用户视图、部门视图添加筛选 |
| IATF | 部门配置 | 迁移 DEPT_CONFIG 到数据库（可选） |

@auth.requires_login()
def department():
    """部门管理页面"""
    _check_admin()
    _setup_page('部门管理')

    # 获取部门列表（带成员统计）
    departments = db(db.department.is_active == True).select(
        orderby=db.department.sort_order
    )

    # 统计每个部门的人数
    dept_counts = {}
    for row in db(db.auth_user.dept_id != None).select(
        db.auth_user.dept_id,
        db.auth_user.id.count().with_alias('count'),
        groupby=db.auth_user.dept_id
    ):
        dept_counts[row.auth_user.dept_id] = row.count

    for dept in departments:
        dept.member_count = dept_counts.get(dept.id, 0)

    return response.render('mgr/department.html', dict(
        departments=departments
    ))


@auth.requires_login()
def department_tree():
    """获取部门树（JSON）"""
    _check_admin()

    departments = db(db.department.is_active == True).select(
        orderby=db.department.sort_order
    )

    # 统计人数
    dept_counts = {}
    for row in db(db.auth_user.dept_id != None).select(
        db.auth_user.dept_id,
        db.auth_user.id.count().with_alias('count'),
        groupby=db.auth_user.dept_id
    ):
        dept_counts[row.auth_user.dept_id] = row.count

    # 构建树形结构
    def build_tree(parent_id=None):
        result = []
        for dept in departments:
            if dept.parent_id == parent_id:
                result.append({
                    'id': dept.id,
                    'name': dept.name,
                    'code': dept.code,
                    'level': dept.level,
                    'parent_id': dept.parent_id,
                    'member_count': dept_counts.get(dept.id, 0),
                    'children': build_tree(dept.id)
                })
        return result

    return response.json({'departments': build_tree()})


@auth.requires_login()
def department_edit():
    """新增/编辑部门"""
    _check_admin()
    dept_id = request.args(0)

    if dept_id:
        dept = db.department(dept_id)
        if not dept:
            raise HTTP(404, "部门不存在")
        form = SQLFORM(db.department, dept_id)
        response.title = f'编辑部门: {dept.name}'
    else:
        form = SQLFORM(db.department)
        response.title = '新增部门'

    if form.process().accepted:
        session.flash = '部门已保存'
        redirect(URL('department'))

    _setup_page(response.title)
    return response.render('mgr/department_edit.html', dict(form=form))


@auth.requires_login()
def department_delete():
    """删除部门"""
    _check_admin()
    dept_id = request.args(0)

    if not dept_id:
        return response.json({'error': '缺少部门ID'})

    dept = db.department(dept_id)
    if not dept:
        return response.json({'error': '部门不存在'})

    # 检查是否有子部门
    children = db(db.department.parent_id == dept_id).count()
    if children > 0:
        return response.json({'error': '该部门下有子部门，无法删除'})

    # 检查是否有成员
    members = db(db.auth_user.dept_id == dept_id).count()
    if members > 0:
        return response.json({'error': f'该部门下有 {members} 名成员，请先转移成员'})

    try:
        # 软删除
        dept.update_record(is_active=False)
        db.commit()
        return response.json({'success': True})
    except Exception as e:
        db.rollback()
        return response.json({'error': str(e)})


@auth.requires_login()
def department_members():
    """获取部门成员"""
    _check_admin()
    dept_id = request.args(0)

    if not dept_id:
        return response.json({'error': '缺少部门ID'})

    dept = db.department(dept_id)
    if not dept:
        return response.json({'error': '部门不存在'})

    members = db(db.auth_user.dept_id == dept_id).select(
        db.auth_user.id,
        db.auth_user.username,
        db.auth_user.first_name,
        db.auth_user.last_name,
        db.auth_user.email,
        orderby=db.auth_user.username
    )

    return response.json({
        'department': {'id': dept.id, 'name': dept.name},
        'members': [dict(m) for m in members],
        'total': len(members)
    })


@auth.requires_login()
def department_stats():
    """部门统计"""
    _check_admin()

    # 各部门人数统计
    stats = []
    for row in db(db.department.is_active == True).select(
        db.department.id,
        db.department.name,
        db.department.code,
        orderby=db.department.sort_order
    ):
        count = db(db.auth_user.dept_id == row.id).count()
        stats.append({
            'id': row.id,
            'name': row.name,
            'code': row.code,
            'member_count': count
        })

    # 未分配部门的用户数
    unassigned = db(
        (db.auth_user.dept_id == None) | (db.auth_user.dept_id == 0)
    ).count()

    return response.json({
        'departments': stats,
        'unassigned': unassigned,
        'total_users': db(db.auth_user).count()
    })

# scripts/migrate_department.py
"""
部门功能迁移脚本
使用方法: python web2py.py -S service_center -M -R scripts/migrate_department.py
"""

def migrate():
    print("开始迁移...")

    # 步骤1: 确认 department 表已创建（由 models/dept_db.py 自动完成）
    print(f"部门表记录数: {db(db.department).count()}")

    # 步骤2: 确认 auth_user.dept_id 字段存在
    if 'dept_id' not in db.auth_user.fields:
        print("警告: auth_user.dept_id 字段不存在，请检查 models/dept_db.py")
        return

    print(f"auth_user 表记录数: {db(db.auth_user).count()}")

    # 步骤3: 统计未分配部门的用户
    unassigned = db(
        (db.auth_user.dept_id == None) | (db.auth_user.dept_id == 0)
    ).count()
    print(f"未分配部门的用户: {unassigned}")

    print("迁移检查完成！")

if __name__ == '__main__' or True:
    migrate()
